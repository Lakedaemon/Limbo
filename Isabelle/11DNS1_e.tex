		\noindent
		Le but de cet exercice est d'étudier la fonction $f$ définie sur 
		$[0,+\infty[$ par :
		$$\left\{ \begin{array}{ll}
		f(x) = \dfrac{x\ln (x)}{x+1} & \text{ si } x > 0 \\
		f(0) = 0 
		\end{array} \right.$$
		\begin{enumerate}
			\item Étudier la continuité et la dérivabilité de $f$ en $0$.
			\item 
				\begin{enumerate}
					\item Soit $\varphi$ la fonction définie sur 
						$]0,+\infty[$ par :
						$\varphi (x) = \ln (x) + x + 1$.\\
						Étudier les variations de $\varphi$.
					\item Établir que l'équation $\varphi (x) = 0$ admet 
						une solution $\beta$ et une seule.
					\item Montrer que 
						$0,27 \leqslant \beta \leqslant 0,28$. \\
						(On ne demande pas de construire la 
						courbe représentative de $\varphi$.)
					\item Pour $x > 0$, exprimer $f'(x)$ à l'aide de 
						$\varphi (x)$.
					\item En déduire les variations de $f$.
				\end{enumerate}
			\item 
				\begin{enumerate}
					\item Déterminer la limite de $f$ en $+\infty$
					\item Déterminer la limite de $\ln (x) - f(x)$ 
						lorsque $x$ tend vers $+\infty$.
				\end{enumerate}
			\item Construire soigneusement les courbes représentatives 
				${\cal C}$ de $f$ et $\Gamma$ de $x \mapsto \ln (x)$ 
				dans un repère orthonormé. \\
				En particulier, on cherchera les annulations de $f$ 
				et on donnera la valeur de $f(\beta)$ à l'aide de 
				$\beta$.\\[2cm]
		\end{enumerate}

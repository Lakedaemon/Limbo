%% process the Limbo xml files
%
% Sectioning : part/chapter/section/subsection
%
% Grouping : concept/group
%
% Statement : definition/notation - property/theorem/corrolary/lemma
%
% Illustrations : figure/example(application)
%
% Explanations : uses/text/remark/method
%
% Prooving: proof/
%
% Exercice: exercise/ 


\startxmlsetups xml:cours
\xmlsetsetup{cours}{*}{-}
% root
\xmlsetsetup{cours}{root}{xml:cours:flush}
% includes
\xmlsetsetup{cours}{include}{xml:cours:include}
\xmlsetsetup{cours}{ref}{xml:cours:inject}
% sectioning
\xmlsetsetup{cours}{domain|chapter|section|subsection|subsubsection|concept}{xml:cours:group}
% grouping 
\xmlsetsetup{cours}{group}{xml:cours:group}
\xmlsetsetup{cours}{combination}{xml:cours:combination}

\xmlsetsetup{cours}{context}{xml:cours:context}
%\xmlsetsetup{cours}{combination/property}{xml:combination/statement}
% statements (primary objects)
\xmlsetsetup{cours}{corollary|definition|lemma|notation|property|theorem|}{xml:cours:statement} 
% best practice (secondary info)
\xmlsetsetup{cours}{craft|uses|item}{xml:cours:*}
% illustration
\xmlsetsetup{cours}{figure|example}{xml:cours:*} 
% text (tertiary info)
\xmlsetsetup{cours}{remark|remarks|text}{xml:cours:*} 
%\xmlsetsetup{cours}{remarks/remark}{xml:remarks:remark} 
% exercises
\xmlsetsetup{cours}{exercise}{xml:cours:*}
% proofs
\xmlsetsetup{cours}{proof}{xml:cours:*}

\stopxmlsetups 

\xmlregistersetup{xml:cours} 

\startxmlsetups xml:cours:flush
\xmlflush{#1}%
\stopxmlsetups

\startxmlsetups xml:cours:craft
\ctxlua{L.processStatement("#1")}%
\stopxmlsetups

% example
\iffalse
\startxmlsetups xml:example
\ctxlua{L.processExample("#1")}%
\stopxmlsetups
\fi


\startxmlsetups xml:cours:include
\ctxlua{L.processInclude("#1")}%
\stopxmlsetups

\startxmlsetups xml:cours:inject
\ctxlua{L.processInject("#1")}%
\stopxmlsetups

\startxmlsetups xml:cours:context
\xmlflush{#1}%
\stopxmlsetups

\startxmlsetups xml:cours:group
\ctxlua{L.processGroup("#1")}%
\stopxmlsetups

\startxmlsetups xml:cours:statement
\ctxlua{L.processNewStatement("#1")}%
%\placeleftfigure{}{\useMPgraphic{Frame}}%
\stopxmlsetups

\startxmlsetups xml:cours:combination
\xmlflush{#1}%
%\ctxlua{L.processCombination("#1")}%
\stopxmlsetups

%exercices
\iffalse
\startxmlsetups xml:exercise
\ctxlua{L.processExercise("#1")}%
\stopxmlsetups
\fi

\startxmlsetups xml:cours:figure
\ctxlua{L.processFigure("#1")}%
\stopxmlsetups

% proofs
\iffalse
\startxmlsetups xml:proof
\ctxlua{L.processProof("#1")}%
\stopxmlsetups
\fi

% remarks
\iffalse
\startxmlsetups xml:remark
\ctxlua{L.processRemark("#1")}%
\stopxmlsetups

\startxmlsetups xml:remarks 
{\tenpoint\startitemize[packed,intext,joinedup,2,nowhite,after]%
\xmlflush{#1}%
\stopitemize}%
\stopxmlsetups

\startxmlsetups xml:remarks:remark
\item\ctxlua{L.processRawContent("#1")}%
\stopxmlsetups
\fi

% texts
\iffalse
\startxmlsetups xml:text

\ctxlua{L.processRawContent("#1")}%
\stopxmlsetups
\fi

\iffalse
\startxmlsetups xml:uses
{\tenpoint
\startitemize[packed,intext,joinedup,4,nowhite,after]
\xmlflush{#1}%
\stopitemize}
\stopxmlsetups


\startxmlsetups xml:item
\ctxlua{L.processContent("#1","\\item")}%
\stopxmlsetups
\fi

\startxmlsetups xml:cours:flushsidefloats
\flushsidefloats
\stopxmlsetups
